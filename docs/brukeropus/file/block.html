 13class FileBlock:
 14    '''Generic OPUS file block.
 15
 16    This class initializes with the most basic file block info from the file directory: type, size, and start location
 17    as well as the raw bytes from the file (which can subsequently be parsed).
 18
 19    Args:
 20        filebytes: raw bytes of the file
 21        block_type: six integer tuple that describes the type of data in the file block
 22        size: size of block in number of bytes
 23        start: pointer to start location of the block within the file
 24
 25    Attributes:
 26        type: six integer tuple that describes the type of data in the file block
 27        size: size of block in number of bytes
 28        start: pointer to start location of the block within the file
 29        bytes: raw bytes of file block (set to zero bytes if successfully parsed)
 30        data: parsed data if successful. Could be: `list`, `str`, `np.ndarray` or `dict` depending on the block type.
 31        parser: name of parsing function if parsing was successful
 32    '''
 33
 34    __slots__ = ('type', 'size', 'start', 'bytes', 'data', 'parser', 'keys')
 35
 36    def __init__(self, filebytes: bytes, block_type: tuple, size: int, start: int):
 37        self.bytes = filebytes[start: start + size]
 38        self.type = block_type
 39        self.size = size
 40        self.start = start
 41        self.data = None
 42        self.parser = None
 43
 44    def __str__(self):
 45        label = self.get_label()
 46        return 'FileBlock: ' + label + ' (size: ' + str(self.size) + ' bytes; start: ' + str(self.start) + ')'
 47
 48    def _try_parser(self, parser):
 49        try:
 50            self.data = parser(self.bytes)
 51            if type(self.data) is dict:
 52                self.keys = list(self.data.keys())
 53            self._clear_parsed_bytes(parser=parser)
 54        except Exception as e:
 55            self.data = 'Error parsing: ' + str(e)
 56
 57    def _clear_parsed_bytes(self, parser):
 58        '''Clear raw bytes that have been parsed (and log the parser for reference)'''
 59        self.parser = parser.__name__
 60        self.bytes = b''
 61
 62    def is_data_status(self):
 63        '''Returns True if `FileBlock` is a data status parameter block'''
 64        return self.type[2] == 1
 65
 66    def is_rf_param(self):
 67        '''Returns True if `FileBlock` is a parameter block associated with the reference measurement (not including
 68        data status blocks)'''
 69        return self.type[2] > 1 and self.type[1] == 2
 70
 71    def is_param(self):
 72        '''Returns True if `FileBlock` is any parameter block (could be data status, rf param, sample param, etc.)'''
 73        return self.type[2] > 0 or self.type == (0, 0, 0, 0, 0, 1)
 74
 75    def is_sm_param(self):
 76        '''Returns True if `FileBlock` is a parameter block associated with sample/result measurement (not including
 77        data status blocks)'''
 78        return self.is_param() and not self.is_data_status() and not self.is_rf_param()
 79
 80    def is_directory(self):
 81        '''Returns True if `FileBlock` is the directory block'''
 82        return self.type == (0, 0, 0, 13, 0, 0)
 83
 84    def is_file_log(self):
 85        '''Returns True if `FileBlock` is the file log (aka 'history') block'''
 86        return self.type == (0, 0, 0, 0, 0, 5)
 87
 88    def is_data(self):
 89        '''Returns True if `FileBlock` is a 1D data block (not a data series)'''
 90        return self.type[2] == 0 and self.type[3] not in [0, 13] and self.type[5] != 2
 91
 92    def is_data_series(self):
 93        '''Returns True if `FileBlock` is a data series block (i.e. 3D data)'''
 94        return self.type[2] == 0 and self.type[5] == 2
 95
 96    def get_label(self):
 97        '''Returns a friendly string label that describes the block type'''
 98        return get_block_type_label(self.type)
 99
100    def get_data_key(self):
101        '''If block is a data block, this function will return a shorthand key to reference that data.
102
103        e.g. t: transmission, a: absorption, sm: sample, rf: reference, phsm: sample phase etc. If the block is not
104        a data block, it will return `None`.'''
105        if self.is_data() or self.is_data_series():
106            return get_data_key(self.type)
107        else:
108            return None
109
110    def get_parser(self):
111        '''Returns the appopriate file block parser based on the type code (None if not recognized)'''
112        if self.is_directory():
113            return parse_directory
114        elif self.is_file_log():
115            return parse_text
116        elif self.is_param():
117            return parse_params
118        elif self.is_data_series():
119            return parse_data_series
120        elif self.is_data():
121            return parse_data
122        else:
123            return None

125    def parse(self):
126        '''Determines the appropriate parser for the block and parses the raw bytes.  Parsed data is stored in `data`
127        attribute and `bytes` attribute is set empty to save memory'''
128        parser = self.get_parser()
129        if parser is not None:
130            self._try_parser(parser)

133class FileDirectory:
134    '''Contains type and pointer information for all blocks of data in an OPUS file.
135
136    `FileDirectory` information is decoded from the raw file bytes of an OPUS file. First the header is read which
137    provides the start location of the directory block, number of blocks in file, and maximum number of blocks the file
138    supports. Then it decodes the block pointer information from each entry of the file's directory block to create a
139    `FileBlock` instance, initiates the block parsing, and adds the parsed block to the `blocks` attribute.
140
141    Args:
142        filebytes: raw bytes from OPUS file. see: `brukeropus.file.parser.read_opus_file_bytes`
143
144    Attributes:
145        start: pointer to start location of the directory block
146        max_blocks: maximum number of blocks supported by file
147        num_blocks: total number of blocks in the file
148        blocks: list of `FileBlock` from the file. The class parses these blocks upon initilization of the class.
149    '''
150    def __init__(self, filebytes: bytes):
151        self.version, self.start, self.max_blocks, self.num_blocks = parse_header(filebytes)
152        size = self.max_blocks * 3 * 4
153        blocks = []
154        for block_type, size, start in parse_directory(filebytes[self.start: self.start + size]):
155            block = FileBlock(filebytes=filebytes, block_type=block_type, size=size, start=start)
156            block.parse()
157            blocks.append(block)
158        self.blocks = blocks

161def is_data_status_type_match(data_block: FileBlock, data_status_block: FileBlock) -> bool:
162    '''Checks if data and data status blocks are a match based soley on the block type.
163
164    This check correctly and accurately matches blocks most of the time, but is occasionally not sufficient on its own
165    (e.g. when multiple spectra of the same exact type are stored in a single file)'''
166    t1 = data_status_block.type
167    t2 = data_block.type
168    return t1[:2] == t2[:2] and t1[3:] == t2[3:]

171def is_data_status_val_match(data_block: FileBlock, data_status_block: FileBlock) -> bool:
172    '''Checks if min(data) and max(data) match up with the data status parameters: MNY and MXY.
173
174    When multiple spectra of the same type exist in a file, this is used to distinguish if the data and data status
175    blocks are a good match.  This can reduce the number of duplicate matches, but is not generally sufficient to
176    fully eliminate duplicate matches.
177    
178    See test file: `Test Vit C_Glass.0000_comp.0`'''
179    if data_block.is_data():
180        try:
181            ds = data_status_block.data
182            if len(data_block.data) < ds['npt']:
183                return False
184            else:
185                y = ds['csf'] * data_block.data[:ds['npt']]
186                return y.min() == ds['mny'] and y.max() == ds['mxy']
187        except:
188            return True # If error, can't rule out the match
189    else:
190        return True # Don't rule out data series at this time (no example files to test)

193def is_valid_match(data_block: FileBlock, data_status_block: FileBlock) -> bool:
194    '''Checks that number of points in data status are less than or equal to length of parsed data block.
195
196    This does not apply to data series. While rare, it is occasionally necessary to remove these bad matches.
197    
198    See test file: `unreadable.0000`'''
199    if data_block.is_data() and len(data_block.data) < data_status_block.data['npt']:
200        return False
201    else:
202        return True

205def pair_data_and_status_blocks(blocks: list) -> list:
206    '''Takes a list of `FileBlock` and returns a list of matching (data, data status) blocks for further processing.
207
208    All valid data blocks have an associated data status parameter block that contains y-scaling and x-axis info.
209    Generally, these blocks can be easily paired with one another by using the block type. However, some files can
210    contain multiple data blocks with the same exact type, which leads to duplicate matches and requires further
211    inspection to accurately pair.  This function uses the following logical sequence to accurately pair these blocks:
212
213        1. Pair by type and isolate singular matches from duplicate matches
214        2. For duplicate matches, check min(data) and max(data) match data status MNY and MXY (including CSF scaling)
215        3. Again isolate singular matches from duplicate matches
216        4. For remaining duplicate matches, remove any matches that are already in the singular match list.
217        5. Again isolate singular matches (ideally no remaining duplicate matches at this point)
218        6. Remove invalid matches from the singular match list if len(data) is < data status NPT (invalid condition)
219        7. Sort the matches in reverse order of where the data blocks are stored in the file (presume last is most
220           recently added, and therefore final spectra which was true for limited test files).
221
222    For the very limited test files available, this is sufficient to accurately pair all the blocks, but it seems
223    plausible that it will not be sufficient for all files that have duplicate data type entries. More test files
224    required for thorough testing.
225    '''
226    data_status = [b for b in blocks if b.is_data_status() and type(b.data) is not str]
227    data = [b for b in blocks if b.is_data() or b.is_data_series() and type(b.data) is not str]
228    type_matches = []
229    for d in data:
230        type_matches.append((d, [b for b in data_status if is_data_status_type_match(d, b)]))
231    single_matches = [(m[0], m[1][0]) for m in type_matches if len(m[1]) == 1]
232    multi_matches = [match for match in type_matches if len(match[1]) > 1]
233    val_matches = []
234    for d, matches in multi_matches:
235        val_matches.append((d, [b for b in matches if is_data_status_val_match(d, b)]))
236    single_matches = single_matches + [(m[0], m[1][0]) for m in val_matches if len(m[1]) == 1]
237    multi_matches = [match for match in val_matches if len(match[1]) > 1]
238    reduced_matches = []
239    single_starts = [m[1].start for m in single_matches]
240    for d, matches in multi_matches:
241        reduced_matches.append((d, [b for b in matches if b.start not in single_starts]))
242    single_matches = single_matches + [(m[0], m[1][0]) for m in reduced_matches if len(m[1]) == 1]
243    multi_matches = [match for match in reduced_matches if len(match[1]) > 1]
244    
245    single_matches = [match for match in single_matches if is_valid_match(match[0], match[1])] # remove invalid
246    
247    single_matches.sort(key=lambda pairs: pairs[0].start, reverse=True) # last spec seems to be OPUS preference
248    return single_matches

def is_data_status_val_match( data_block: FileBlock, data_status_block: FileBlock) -> bool: 
172def is_data_status_val_match(data_block: FileBlock, data_status_block: FileBlock) -> bool:
173    '''Checks if min(data) and max(data) match up with the data status parameters: MNY and MXY.
174
175    When multiple spectra of the same type exist in a file, this is used to distinguish if the data and data status
176    blocks are a good match.  This can reduce the number of duplicate matches, but is not generally sufficient to
177    fully eliminate duplicate matches.
178    
179    See test file: `Test Vit C_Glass.0000_comp.0`'''
180    if data_block.is_data():
181        try:
182            ds = data_status_block.data
183            if len(data_block.data) < ds['npt']:
184                return False
185            else:
186                y = ds['csf'] * data_block.data[:ds['npt']]
187                return y.min() == ds['mny'] and y.max() == ds['mxy']
188        except:
189            return True # If error, can't rule out the match
190    else:
191        return True # Don't rule out data series at this time (no example files to test)

def pair_data_and_status_blocks(blocks: list) -> list: 
206def pair_data_and_status_blocks(blocks: list) -> list:
207    '''Takes a list of `FileBlock` and returns a list of matching (data, data status) blocks for further processing.
208
209    All valid data blocks have an associated data status parameter block that contains y-scaling and x-axis info.
210    Generally, these blocks can be easily paired with one another by using the block type. However, some files can
211    contain multiple data blocks with the same exact type, which leads to duplicate matches and requires further
212    inspection to accurately pair.  This function uses the following logical sequence to accurately pair these blocks:
213
214        1. Pair by type and isolate singular matches from duplicate matches
215        2. For duplicate matches, check min(data) and max(data) match data status MNY and MXY (including CSF scaling)
216        3. Again isolate singular matches from duplicate matches
217        4. For remaining duplicate matches, remove any matches that are already in the singular match list.
218        5. Again isolate singular matches (ideally no remaining duplicate matches at this point)
219        6. Remove invalid matches from the singular match list if len(data) is < data status NPT (invalid condition)
220        7. Sort the matches in reverse order of where the data blocks are stored in the file (presume last is most
221           recently added, and therefore final spectra which was true for limited test files).
222
223    For the very limited test files available, this is sufficient to accurately pair all the blocks, but it seems
224    plausible that it will not be sufficient for all files that have duplicate data type entries. More test files
225    required for thorough testing.
226    '''
227    data_status = [b for b in blocks if b.is_data_status() and type(b.data) is not str]
228    data = [b for b in blocks if b.is_data() or b.is_data_series() and type(b.data) is not str]
229    type_matches = []
230    for d in data:
231        type_matches.append((d, [b for b in data_status if is_data_status_type_match(d, b)]))
232    single_matches = [(m[0], m[1][0]) for m in type_matches if len(m[1]) == 1]
233    multi_matches = [match for match in type_matches if len(match[1]) > 1]
234    val_matches = []
235    for d, matches in multi_matches:
236        val_matches.append((d, [b for b in matches if is_data_status_val_match(d, b)]))
237    single_matches = single_matches + [(m[0], m[1][0]) for m in val_matches if len(m[1]) == 1]
238    multi_matches = [match for match in val_matches if len(match[1]) > 1]
239    reduced_matches = []
240    single_starts = [m[1].start for m in single_matches]
241    for d, matches in multi_matches:
242        reduced_matches.append((d, [b for b in matches if b.start not in single_starts]))
243    single_matches = single_matches + [(m[0], m[1][0]) for m in reduced_matches if len(m[1]) == 1]
244    multi_matches = [match for match in reduced_matches if len(match[1]) > 1]
245    
246    single_matches = [match for match in single_matches if is_valid_match(match[0], match[1])] # remove invalid
247    
248    single_matches.sort(key=lambda pairs: pairs[0].start, reverse=True) # last spec seems to be OPUS preference
249    return single_matches
